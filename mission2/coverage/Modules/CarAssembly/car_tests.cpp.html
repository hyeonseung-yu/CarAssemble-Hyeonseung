TEST(Assembler_TEST, ValidAssemblyTest) {
	Car car;
	Engine engine;
	Brake brake;
	Steering steering;
	Assembler& assembler = Assembler::getInstance();
	
	car.setType(SEDAN);
	brake.setType(CONTINENTAL);
	bool ret = assembler.isValidAssembly(car, engine, brake, steering);
	EXPECT_FALSE(ret);

	car.setType(SUV);
	engine.setType(TOYOTA);
	ret = assembler.isValidAssembly(car, engine, brake, steering);
	EXPECT_FALSE(ret);

	car.setType(TRUCK);
	engine.setType(WIA);
	ret = assembler.isValidAssembly(car, engine, brake, steering);
	EXPECT_FALSE(ret);

	car.setType(TRUCK);
	engine.setType(TOYOTA);
	brake.setType(MANDO);
	ret = assembler.isValidAssembly(car, engine, brake, steering);
	EXPECT_FALSE(ret);

	car.setType(SUV);
	engine.setType(GM);
	brake.setType(BOSCH_B);
	steering.setType(MOBIS);
	ret = assembler.isValidAssembly(car, engine, brake, steering);
	EXPECT_FALSE(ret);

	car.setType(SUV);
	engine.setType(GM);
	brake.setType(MANDO);
	steering.setType(BOSCH_S);
	ret = assembler.isValidAssembly(car, engine, brake, steering);
	EXPECT_TRUE(ret);
}
